1/*Class to statically condense and recover the following system:
2 |K G||d|=|F|
3 |G^T M||s|=|H|
4 where sizes are as follows: K(dof1,dof1), G(dof1,dof2), M(dof2,dof2), F(dof1), H(dof2)
5
6 Call to staticCondense(), constructs K2, F2:
7 K2=K-G*inv(M)*G^T
8 F2=F-G*inv(M)*H
9
10 And call to recover(), computes s:
11 s = inv(M)*(H-G^T*d)
12*/
13
14/*
15 NOTE: initialize the elemental level staticCondensation class objects by including the following line in the elasticity class definition (or whatever your main problem class is):
16 std::vector<staticCondensation<dof1,dof2> > staticCondensationData;
17 where dof1, dof2 are integer values defining the size of K,G,M,F,H. (For example - std::vector<staticCondensation<3,1> > staticCondensationData;)
18
19 Then include the following line in setup_system():
20 staticCondensationData.resize(triangulation_cp.n_active_cells());
21
22 Now you can access the elemental level staticCondensation object inside the assemble_system() function. For example, access the K matrix or print the F vector of 10th element as follows:
23 staticCondensationData[9].K;
24 staticCondensationData[9].F.print();
25*/

80template <int _dof1, int _dof2>
81void staticCondensation<_dof1,_dof2>::staticCondense(){
82 FullMatrix<double> invM(dof2,dof2), GinvM(dof1,dof2), GinvMG(dof1,dof1);
83 Vector <double> GinvMH(dof1);
84 invM.invert(M); //invM=inverse(M)
85 G.mmult(GinvM,invM); //GinvM=G*inv(M)
86 GinvM.mTmult(GinvMG, G); //GinvMG=G*inv(M)*G^T
87 GinvM.vmult(GinvMH, H); //GinvMH=G*inv(M)*H
88 //compute K2=K-G*inv(M)*G^T
89 K2=K;
90 K2.add(-1.0, GinvMG);
91 //compute F2=F-G*inv(M)*H
92 F2=F;
93 F2.add(-1.0,GinvMH);
94}

98template <int _dof1, int _dof2>
99void staticCondensation<_dof1,_dof2>::recover(){
100 FullMatrix<double> invM(dof2,dof2), invMG(dof2,dof1);
101 Vector <double> invMH(dof2), invMGd(dof2);
102 invM.invert(M); //invM=inverse(M)
103 invM.vmult(invMH,H); //invMH=invM*H
104 invM.mTmult(invMG,G); //invMG=inv(M)*G^T
105 invMG.vmult(invMGd,d); //invMGd=invMG*d
106 //compute s = inv(M)*(H-G^T*d)
107 s=0;
108 s.add(-1.0, invMH, -1.0, invMGd);
109}

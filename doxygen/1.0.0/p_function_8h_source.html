1// pFunction: An interface for PFunctions in PRISMS-PF
2// This class is used to declare and load PFunctions from a pre-existing PLibrary
3// that was created using PRISMS IntegrationTools. The class is a wrapper for the
4// PFunctions in IntegrationTools so that they can be used for vectorized arrays.
5// This class also calls the IntegrationTools checkout function in the constructor
6// to reduce the number of steps the user needs to take. Currently this is only
7// implemented for scalar functions. Vector functions can be treated component by
8// component.
9
10namespace PFunctions{
11
12class pFunction
13{
14public:
15 // Constructor, wraps the IntegrationTools checkout function
16 pFunction(std::string function_name);
17
18 // Returns the value of the function for a given input variable
19 scalarvalueType val(scalarvalueType);
20
21 // Returns one of first derivatives of the function for a given input variable
22 scalarvalueType grad(scalarvalueType, unsigned int);
23
24 // Returns one of the second derivatives of the function for a given input variable
25 scalarvalueType hess(scalarvalueType, unsigned int, unsigned int);
26
27private:
28 PRISMS::PFunction<double*, double> fun;
29};
30
31pFunction::pFunction(std::string function_name){

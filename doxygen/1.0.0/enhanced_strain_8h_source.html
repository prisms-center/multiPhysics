218template <int dim>
219Tensor<1,dim,double> enhancedStrain<dim>::tilde_grad(unsigned int function_no, unsigned int point_no){
220 //Enhanced strain gradient functions for the four incompatible shape functions
221
222 //Ensure the initiliaziation function has been called
223 if(enh_dofs_initialized == false){
224 *pcout << "Error: enhanced dofs not yet initialized.\n";
225 exit(1);
226 }
227
228 //Tensor holding values of the specified enhanced strain gradient function at the given quad. point
229 Tensor<1,dim,double> grad; grad = 0.;
230 //Jacobian at quadrature point
231 Tensor<2,dim,double> J_xi = fe_values.jacobian(point_no);
232 double j_xi = determinant(J_xi);
233
234 //Get the coordinates of the quadrature point in the unit domain
235 Point<dim,double> xi = fe_values.get_quadrature().point(point_no);
236 //Convert xi to the bi-unit domain (deal.II uses a unit domain)
237 for(unsigned int i=0; i<dim; i++){
238 xi(i) = 2.*xi(i) - 1.;
239 }
240
241 //Define enhanced shape function gradients (see paper cited at top)
242 if(function_no < 3*dim){
243 //\tilde{N^i} = 0.5*(xi(i)^2 - 1), i = 0,1,2, from functions listed
244 //in eqn (3.8) of the cited paper
245 grad[function_no/dim] = xi(function_no/dim);
246 }
247 else if(function_no >= 3*dim && function_no < 4*dim){
248 //\tilde{N^3} = xi(0)*xi(1)*xi(2), see equation (3.17) from cited paper
249 grad[0] = xi(1)*xi(2);
250 grad[1] = xi(0)*xi(2);
251 grad[2] = xi(0)*xi(1);
252 }
253 else{
254 *pcout << "Error: invalid function number.\n";
255 exit(1);
256 }
257
258 //See the 2nd line of equation (4.1) of the cited paper
259 grad = j_0/j_xi*grad*J_0inv;
260
261 return grad;
262}

264template <int dim>
265Tensor<1,dim,double> enhancedStrain<dim>::bar_grad(unsigned int function_no, unsigned int point_no,Tensor<1,dim,double> alpha_4){
266 //Enhanced strain gradient functions to replace the standard basis function gradients
267
268 if(enh_dofs_initialized == false){
269 *pcout << "Error: enhanced dofs not yet initialized.\n";
270 exit(1);
271 }
272
273 //Hold enhanced strain gradient function value
274 Tensor<1,dim,double> grad;
275
276 //Jacobian at given quad. point
277 Tensor<2,dim> J_xi = fe_values.jacobian(point_no);
278 double j_xi = determinant(J_xi);
279
280 //Modified basis function gradient to replace standard basis function gradient
281 //See line 3 of equation (4.1) of paper cited at top, as well as (2.17).
282 //Note that deal.II's basis function gradients are all w.r.t. the actual domain,
283 //not the bi-unit or unit domain.
284 grad = (1. + alpha_4*tilde_grad(3*dim,point_no))*center_grad(function_no);
285 grad += j_0/j_xi*(fe_values.shape_grad(function_no,point_no) - center_grad(function_no))*J_xi*J_0inv;
286
287 return grad;
288}

315template <int dim>
316void enhancedStrain<dim>::get_F_enh(unsigned int q, FullMatrix<double> &F){
317
318 if(enh_dofs_initialized == false){
319 *pcout << "Error: enhanced dofs not yet initialized.\n";
320 exit(1);
321 }
322
323 //Get the number of standard dofs and primary enhanced dofs per element
324 unsigned int dofs_per_elem = fe_values.get_fe().dofs_per_cell;
325 unsigned int prim_enh_dofs_per_elem = primary_enhanced_dofs_per_cell();
326
327 //Create vector of original location of each standard dof in the element
328 //This location is the x, y, or z location based on the orientation of the dof.
329 MappingQ1<dim,dim> mapping;
330 std::vector<double> nodalCoords (dofs_per_elem);
331 for (unsigned int d=0; d<dofs_per_elem; ++d){
332 //Check the orientation of the standard dof (displacement in the x, y, or z direction)
333 unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
334 //Get the corresponding component of the position of the node in the initial configuration,
335 //i.e. if the dof is displacement in the y direction at a particular node, store the y component
336 //of that node's location.
337 nodalCoords[d] = mapping.transform_unit_to_real_cell(currentElem,fe_values.get_fe().unit_support_point(d))[i];
338 }
339
340 //define F_tilde (based on enhanced dofs)
341 //see the 2nd half of equation (4.2) of the paper
342 Tensor<2,dim,double> F_tilde; F_tilde = 0.0;
343 for (unsigned int A=0; A<prim_enh_dofs_per_elem; ++A){
344 unsigned int i = enhanced_system_to_component_index(A);
345 F_tilde[i] += Alpha(4*dim*currentElem->user_index()+A)*tilde_grad(A,q);
346 }
347
348 //The fourth alpha vector
349 Tensor<1,dim,double> alpha_4;
350 for (unsigned int i=0; i<dim; ++i){
351 alpha_4[i] = Alpha(4*dim*currentElem->user_index()+9+i);
352 }
353
354 //define F_bar, F_0 (based on standard dofs)
355 //see the 1st half of equation (4.2) of the paper
356 Tensor<2,dim,double> F_bar; F_bar = 0.0;
357 for (unsigned int d=0; d<dofs_per_elem; ++d){
358 unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
359 //F is the derivative of (displacement + original position = deformed position)
360 F_bar[i] += (Ulocal[d] + nodalCoords[d])*bar_grad(d,q,alpha_4);
361 }
362
363 //define F_enhanced = GRADbar_{X}[phi] + GRADtilde_{X}[phi]
364 //equation (4.2) of the paper
365 Tensor<2,dim,double> F_enh; F_enh = 0.0;
366 F_enh = F_bar + F_tilde;
367
368 //Copy F_enh into the user's F matrix
369 F.copy_from(F_enh);
370}

372template <int dim>
373void enhancedStrain<dim>::create_block_mat_vec(FullMatrix<double> F, FullMatrix<double> tau, Tensor<4,dim,double> c_ep, unsigned int q){
374 //Refer to the paper cited at the top
375
376 if(enh_dofs_initialized == false){
377 *pcout << "Error: enhanced dofs not yet initialized.\n";
378 exit(1);
379 }
380
381 Klocal = 0.; Glocal = 0.; Mlocal = 0.;
382 Flocal = 0.; Hlocal = 0.; HlocalT = 0.;
383
384 Tensor<2,dim,double> F_enh, Fenh_inv, tau_enh;
385 unsigned int dofs_per_elem = fe_values.get_fe().dofs_per_cell;
386 unsigned int prim_enh_dofs_per_elem = primary_enhanced_dofs_per_cell();
387 unsigned int enh_dofs_per_elem = enhanced_dofs_per_cell();
388
389 //Copy inputs to the class variables
390 F.copy_to(F_enh);
391 tau.copy_to(tau_enh);
392 Fenh_inv = invert(F_enh);
393
394 //Retrieve the 4th Alpha vector for this element
395 Tensor<1,dim,double> alpha_4;
396 for (unsigned int i=0; i<dim; ++i){
397 alpha_4[i] = Alpha(4*dim*currentElem->user_index()+9+i);
398 }
399
400 //Create vector of original location of each standard dof in the element
401 //This location is the x, y, or z location based on the orientation of the dof.
402 MappingQ1<dim,dim> mapping;
403 std::vector<double> nodalCoords (dofs_per_elem);
404 Tensor<2,dim,double> F_0; F_0 = 0.0;
405 for (unsigned int d=0; d<dofs_per_elem; ++d){
406 //Check the orientation of the standard dof (displacement in the x, y, or z direction)
407 unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
408 //Get the corresponding component of the position of the node in the initial configuration,
409 //i.e. if the dof is displacement in the y direction at a particular node, store the y component
410 //of that node's location.
411 nodalCoords[d] = mapping.transform_unit_to_real_cell(currentElem,fe_values.get_fe().unit_support_point(d))[i];
412 //F is the derivative of (displacement + original position = deformed position)
413 F_0[i] += (Ulocal[d] + nodalCoords[d])*center_grad(d);
414 }
415
416 //Transform enhanced gradients to current configuration
417 //see equation (4.3) of the cited paper
418 std::vector<Tensor<1,dim,double> > bar_del(dofs_per_elem),
419 center_del(dofs_per_elem),
420 tilde_del(prim_enh_dofs_per_elem);
421 for (unsigned int d=0; d<dofs_per_elem; ++d){
422 bar_del[d] = bar_grad(d,q,alpha_4)*Fenh_inv;
423 center_del[d] = center_grad(d)*Fenh_inv;
424 }
425 for (unsigned int d=0; d<prim_enh_dofs_per_elem; d++){
426 tilde_del[d] = tilde_grad(d,q)*Fenh_inv;
427 }
428 Tensor<2,dim,double> F0_Finv;
429 F0_Finv = F_0*Fenh_inv;
430
431 //Evaluate H = F_intTilde_elem
432 for (unsigned int d=0; d<enh_dofs_per_elem; d++){
433 unsigned int i = enhanced_system_to_component_index(d);
434 if(d<prim_enh_dofs_per_elem){
435 //see line 2 of equation (4.9)
436 Hlocal(d) += tau_enh[i]*tilde_del[d];
437 }
438 else{
439 //see line 2 of equation (4.9)
440 Hlocal(d) += scalar_product(F0_Finv,tau_enh)*tilde_grad(d,q)[i]; //double_contract wants symmetricTensor
441 }
442 }
443
444 //Evaluate F = F_intBar_elem
445 for (unsigned int d=0; d<dofs_per_elem; ++d){
446 unsigned int i = fe_values.get_fe().system_to_component_index(d).first;
447 //see line 1 of equation (4.9)
448 Flocal(d) += tau_enh[i]*bar_del[d];
449 }
450
451 //Calculate M (material and geometric parts)
452 for (unsigned int I=0; I<enh_dofs_per_elem; ++I){
453 unsigned int j = enhanced_system_to_component_index(I);
454 for (unsigned int J=0; J<enh_dofs_per_elem; ++J){
455 unsigned int k = enhanced_system_to_component_index(J);
456 for (unsigned int i=0; i<dim; ++i){
457 for (unsigned int l=0; l<dim; ++l){
458 if(I < prim_enh_dofs_per_elem && J < prim_enh_dofs_per_elem){
459 //See line 3 of equations (4.14) and (4.18)
460 Mlocal[I][J] += tilde_del[I][i]*(tau_enh[i][l]*(j==k) + c_ep[i][j][k][l])*tilde_del[J][l];
461 }
462 else if(I < prim_enh_dofs_per_elem){
463 //See also line 2 of equation (4.19)
464 Mlocal[I][J] += (tilde_del[I][i]*tau_enh[i][l]*F0_Finv[j][l] + F0_Finv[i][l]*c_ep[i][l][j]*tilde_del[I])*
465 tilde_grad(J,q)[k];
466 }
467 }
468 if(I < prim_enh_dofs_per_elem && J >= prim_enh_dofs_per_elem){
469 Mlocal[J][I] = Mlocal[I][J];
470 }
471 }
472 //See also line 3 of equation (4.19)
473 if(I >= prim_enh_dofs_per_elem && J >= prim_enh_dofs_per_elem){
474 Mlocal[I][J] += (contract3(F0_Finv,c_ep,F0_Finv) + scalar_product(F0_Finv,tau_enh*F0_Finv))*
475 tilde_grad(I,q)[j]*tilde_grad(J,q)[k];
476 }
477 }
478 }
479
480 //Fill in local K matrix (both material and geometric parts)
481 for (unsigned int I=0; I<dofs_per_elem; ++I){
482 unsigned int j = fe_values.get_fe().system_to_component_index(I).first;
483 for (unsigned int J=0; J<dofs_per_elem; ++J){
484 unsigned int k = fe_values.get_fe().system_to_component_index(J).first;
485 for (unsigned int i=0; i<dim; ++i){
486 for (unsigned int l=0; l<dim; ++l){
487 //See line 1 of equations (4.14) and (4.18)
488 Klocal[I][J] += bar_del[I][i]*(c_ep[i][j][k][l] + tau_enh[i][l]*(j==k))*bar_del[J][l];
489 }
490 }
491 }
492 }
493
494 //Fill in local G matrix (both material and geometric parts)
495 for (unsigned int I=0; I<dofs_per_elem; ++I){
496 unsigned int j = fe_values.get_fe().system_to_component_index(I).first;
497 for (unsigned int J=0; J<enh_dofs_per_elem; ++J){
498 unsigned int k = enhanced_system_to_component_index(J);
499 for (unsigned int i=0; i<dim; ++i){
500 for (unsigned int l=0; l<dim; ++l){
501 if(J<prim_enh_dofs_per_elem){
502 //See line 2 of equations (4.14) and (4.18)
503 Glocal[I][J] += bar_del[I][i]*(c_ep[i][j][k][l] + tau_enh[i][l]*(j==k))*tilde_del[J][l];
504 }
505 else{
506 //See also line 1 of equation (4.19)
507 Glocal[I][J] += (F0_Finv[i][l]*c_ep[i][l][j]*bar_del[I] + (F0_Finv[j][i]*bar_del[I][l] + (j==i)*center_del[I][l])*tau_enh[i][l])*
508 tilde_grad(J,q)[k];
509 }
510 }
511 }
512 }
513 }

518template <int dim>
519void enhancedStrain<dim>::updateAlpha(Vector<double> dUlocal, unsigned int cellID){
520
521 if(enh_dofs_initialized == false){
522 *pcout << "Error: enhanced dofs not yet initialized.\n";
523 exit(1);
524 }
525
526 //loop over elements, pass the standard dofs (local displacment) to static condensation
527 const unsigned int dofs_per_elem = fe_values.get_fe().dofs_per_cell;
528 for (unsigned int n=0; n<dofs_per_elem; ++n){
529 staticCondensationData[cellID].d(n) = dUlocal[n];
530 }
531
532 //Compute delta_Alpha useing static condensation object
533 staticCondensationData[cellID].recover();
534
535 //Add delta_Alpha to Alpha to update the enhanced dofs
536 for(unsigned int n=0; n<4*dim; n++){
537 Alpha(4*dim*cellID+n) += staticCondensationData[cellID].s(n);
538 }
539}

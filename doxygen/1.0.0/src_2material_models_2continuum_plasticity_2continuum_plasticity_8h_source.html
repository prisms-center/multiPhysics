165template <int dim>
166void continuumPlasticity<dim>::init(unsigned int num_quad_points)
167{
168 //Get the total numbers of elements for this processor (num_local_cells)
169 unsigned int num_local_cells = this->triangulation.n_locally_owned_active_cells();
170 //Initiate the enhanced strain object with the number elements
171 enhStrain.init_enh_dofs(num_local_cells);
172
173 //Resize the deformation gradient and Kirchhoff stress tensors
174 F.reinit(dim, dim);
175 tau.reinit(dim, dim);
176
177 //Checkout the elastic strain energy density function from the pfunction library.
178 //These are located in the "models" folder. First check that an available
179 //model was requested.
180 if(properties.strainEnergyModel == "quadlog" ||
181 properties.strainEnergyModel == "neohook" ||
182 properties.strainEnergyModel == "stvenkir"){
183 PRISMS::PLibrary::checkout(properties.strainEnergyModel, strain_energy);
184 }
185 else{
186 this->pcout << "Using user defined strain energy density function.\n";
187 PRISMS::PLibrary::checkout(properties.strainEnergyModel, strain_energy);
188 }
189
190 /*strain energy function inputs:
191 0: lambda, "First Lame parameter"
192 1: mu, "Second Lame parameter"
193 2: lambda1, "First principle stretch"
194 3: lambda2, "Second principle stretch"
195 4: lambda3, "Third principle stretch"*/
196
197 //Resize the vector of parameters/variables used in the strain energy function
198 varStrainEnergy.resize(5,1.);
199 //Specify the material parameters used in the strain energy function
200 varStrainEnergy[0] = properties.lambda;
201 varStrainEnergy[1] = properties.mu;
202
203 //For now, specify the hardening model here and check it out from the
204 //pfunction library (NOTE: this calculates the value for "q", the conjugate
205 // stress-like quantity, used in the yield function).
206 //See, for example, the end of section 2.1 of the formulation.
207 if(properties.isoHardeningModel == "linear_hardening"){
208 PRISMS::PLibrary::checkout(properties.isoHardeningModel, harden);
209 }
210 else{
211 this->pcout << "Using user defined isotropic function.\n";
212 PRISMS::PLibrary::checkout(properties.isoHardeningModel, harden);
213 }
214
215 /*hardening function inputs:
216 0: alpha, "Equivalent plastic strain"
217 1: K, "Hardening parameter"*/
218
219 //Resize the vector of parameters/variables used in the hardening function
220 varIsoHardening.resize(2,0.);
221 //Specify the material parameter used in the hardening function
222 varIsoHardening[1] = properties.K;
223
224 //Checkout the yield function from the pfunction library.
225 //This is located in the "models" folder. First check that an available
226 //model was requested.
227 //See equation (16) of the formulation.
228 if(properties.yieldModel == "von_mises"){
229 PRISMS::PLibrary::checkout(properties.yieldModel, yield);
230 }
231 else{
232 this->pcout << "Using user defined yield function.\n";
233 PRISMS::PLibrary::checkout(properties.yieldModel, yield);
234 }
235
236 /*yield function inputs:
237 0: beta1, "First principle stress"
238 1: beta2, "Second principle stress"
239 2: beta3, "Third principle stress"
240 3: tau_y, "Yield stress"
241 4: q, "Conjugate stress-like quantity"*/
242
243 //Resize the vector of parameters/variables used in the yield function
244 varYield.resize(8,0.);
245 //Specify the material parameter used in the yield function
246 varYield[3] = properties.tau_y;
247
248 //Resize the vectors of history variables according to the number of elements and quadrature points
249 Vector<double> zero_vec(dim); zero_vec = 0.;
250 histInvCP_conv.resize(num_local_cells,std::vector<FullMatrix<double> >(num_quad_points,IdentityMatrix(dim)));
251 histAlpha_conv.resize(num_local_cells,std::vector<double>(num_quad_points,0));
252 histXi_conv.resize(num_local_cells,std::vector<Vector<double> >(num_quad_points,zero_vec));
253 histInvCP_iter.resize(num_local_cells,std::vector<FullMatrix<double> >(num_quad_points,IdentityMatrix(dim)));
254 histAlpha_iter.resize(num_local_cells,std::vector<double>(num_quad_points,0));
255 histXi_iter.resize(num_local_cells,std::vector<Vector<double> >(num_quad_points,zero_vec));
256
257 //Resize the vector of vector used to store the von Mises stress
258 projectVonMisesStress.resize(num_local_cells,std::vector<double>(num_quad_points,0));
259
260 //Initialize the "plasticOnset" marker
261 plasticOnset = false;
262 //Change "initCalled" to "true" to show that the "init" function has now been called
263 initCalled = true;
264}

266template <int dim>
267void continuumPlasticity<dim>::calculatePlasticity(unsigned int cellID,
268 unsigned int quadPtID)
269{
270 //Actual and trial values of the equivalent plastic strain, respectively
271 double alpha, alpha_TR;
272 //Actual and trial values of the back stress vector, respectively
273 Vector<double> xi(dim), xi_TR(dim);
274 //Inverse deformation gradient and elastic left C-G tensor, repsectively
275 FullMatrix<double> F_inv(dim,dim), b_e(dim,dim);
276 //Trial and actual values of the inverse of plastic right C-G tensor, respectively
277 FullMatrix<double> invCp_TR(dim,dim), invCp(dim,dim);
278 //Isotropic tensor (identity matrix)
279 FullMatrix<double> Identity(dim,dim);
280 //Initialize tensors to zero
281 F_inv=0; c=0; invCp=0; tau=0; b_e=0;
282 //Initialize isotropic tensor to correct values.
283 Identity = IdentityMatrix(dim);
284
285 //Set the trial values of invCP, alpha, and xi for the current increment equal to the
286 //converged values of invCP and alpha from the previous increment
287 //see equations (23) and (24) of the formulation
288 invCp_TR = histInvCP_conv[cellID][quadPtID];
289 alpha_TR = histAlpha_conv[cellID][quadPtID];
290 xi_TR = histXi_conv[cellID][quadPtID];
291
292 //Initialize the "identity vector", i.e. the vector [1,1,1] in 3D
293 Vector<double> Ident_vec(3); Ident_vec = 0.; Ident_vec.add(1.);
294 //Find the inverse of the deformation gradient
295 F_inv.invert(F);
296
297 //Elastic trial left C-G tensor (b_eTR) at n+1
298 FullMatrix<double> b_eTR(dim,dim), invCp_TR_FT(dim,dim), be_FinvT(dim,dim);
299 //invCp_FT = invCp_TR*F^T (temporary variable)
300 invCp_TR.mTmult(invCp_TR_FT,F);
301 //b_eTR = F*invCp_TR*F^T (see equation (24) of the formulation)
302 F.mmult(b_eTR,invCp_TR_FT);
303
304 //Compute the eigenvalues/eigenvectors of b_eTR. These are used in the spectral
305 //decomposition of b_e and tau. See equations (27)-(29).
306
307 //Eigenvalues of b_TR^e are square of trial elastic stretches.
308 //Compare to equation (28) of the formulation
309 Vector<double> eigTR(dim);
310 //Some temporary objects are used in the computation.
311 LAPACKFullMatrix<double> b_eTReig(dim,dim), Ident(dim,dim);
312 std::vector< Vector<double> > eigVec(dim);
313 b_eTReig = b_eTR;
314 Ident = Identity;
315 /* if(this->currentIncrement_cp == 1){
316 for(int i=0;i<3;i++){
317 for(int j=0;j<3;j++){
318 this->pcout << invCp_TR[i][j] << " ";
319 }
320 this->pcout << std::endl;
321 }
322 this->pcout << std::endl;
323 }*/
324
325 //Use LAPACK to solve for the eigenvalues/eigenvectors.
326 b_eTReig.compute_generalized_eigenvalues_symmetric(Ident,eigVec);
327 //The dyadic product of the eigenvectors is used in the spectral
328 //decomposition. Again, see equations (28) and (29).
329 std::vector< FullMatrix<double> > eigDyad(dim);
330 for(unsigned int i=0; i<dim; i++){
331 eigTR(i) = abs(b_eTReig.eigenvalue(i));
332 eigDyad[i].outer_product(eigVec[i],eigVec[i]);
333 }
334
335 //These variables are used in the elastoplastic tangent. See equation (42).
336 Vector<double> d_A(dim); d_A = 0.;
337 for(unsigned int i=0; i<dim; i++){
338 d_A(i) = 1.;
339 for(unsigned int j=1; j<dim; j++){
340 //Since we will divide by d_A, we need to be careful if it is almost zero.
341 if(std::abs(eigTR(i) - eigTR((i+j)%3)) < 1.e-13){
342 d_A(i) *= copysign(5.e-8,eigTR(i) - eigTR((i+j)%3));
343 }
344 else{
345 d_A(i) *= eigTR(i) - eigTR((i+j)%3);
346 }
347 }
348 }
349
350 //The principal elastic stretches
351 Vector<double> lambda_e(dim);
352 //The principal stresses and the derivative of the yield function w.r.t. the principal stresses
353 Vector<double> beta(dim), nuBar(dim);
354 for(unsigned int i=0; i<dim; i++){
355 //The principal elastic stretches, lambda_e(i)
356 varStrainEnergy[2+i] = sqrt(eigTR(i));
357 lambda_e[i] = sqrt(eigTR(i));
358 }
359 double gamma_Dt = 0.;
360 for(unsigned int i=0; i<dim; i++){
361 //Calculate the principal stresses, beta, using equation (32)
362 varYield[i] = strain_energy.grad(varStrainEnergy,2+i)*varStrainEnergy[2+i]; //Beta
363 beta(i) = varYield[i];
364 }
365 //Initially take alpha as alpha_TR
366 alpha = alpha_TR;
367 //Initially take xi as xi_TR
368 xi = xi_TR;
369 //alpha is the first variable needed in the hardening function
370 varIsoHardening[0] = alpha;
371 //The variable, q, needed in the yield function gets its value from the hardening function
372 varYield[4] = harden(varIsoHardening);
373 //The back stress is needed for the yield function
374 for(unsigned int i=0; i<dim; i++){
375 varYield[dim+2+i] = xi(i);
376 }
377 //Get the value of the yield function, based on the trial values
378 double yield_TR = yield(varYield);
379 //if(cellID == 0)
380 // std::cout << "Trial yield: " << yield(varYield) << std::endl;
381
382 //Used as a fifth unknown in the nonlinear solve, zeta is to be equal to (df/dq)_{n+1}
383 double zeta = yield.grad(varYield,4);
384
385 //If the yield function is greater than zero, plastic flow has occured.
386 //We will need to update the plastic variables (see section 3 of the formulation) using Newton-Raphson.
387 //The independent variables that we solve for are the 3 principal elastic stretches (lambda_e(i))
388 //and the equivalent plastic strain (alpha). The four equations used are equation (30) and a variation
389 //of equation (26).
390 if(yield_TR > 0){
391 //Report onset of plasticity
392 if(plasticOnset == false){
393 this->pcout << "\ncontinuumPlasticity: Onset of plasticity\n\n";
394 plasticOnset = true;
395 }
396
397 //Being careful dividing by zero...
398 if(alpha_TR == 0){
399 varIsoHardening[0] = 1.e-4; //So that the derivativate of the hardening function isn't undefined.
400 }
401 //Set the tolerance for convergence
402 const double Tolerance = 1.e-12;
403 //Declare the jacobian and residual to solve for the plastic variables.
404 //(product = inv(jacobian)*residual)
405 Vector<double> residual(2+dim), product(2+dim);
406 FullMatrix<double> jacobian(2+dim,2+dim);
407 unsigned int iter=0;
408 //Iterate until convergence is met
409 while(true){
410
411 //Fill in Residual vector and Jacobian matrix for current time step
412 //(Note: the derivation of the jacobian isn't included in the formulation)
413 jacobian = 0.;
414 for (unsigned int B=0; B<dim; ++B){
415 for (unsigned int A=0; A<dim; ++A){
416 for (unsigned int C=0; C<dim; ++C){
417 jacobian[A][B] -= gamma_Dt*yield.hess(varYield,A,C)*(strain_energy.grad(varStrainEnergy,2+C)*(B==C) +
418 strain_energy.hess(varStrainEnergy,2+B,2+C)*varStrainEnergy[2+C]);
419 }
420 jacobian[A][B] -= 1./varStrainEnergy[2+A]*(A==B);
421 jacobian[3][A] += yield.grad(varYield,B)*(strain_energy.grad(varStrainEnergy,2+B)*(A==B) +
422 strain_energy.hess(varStrainEnergy,2+A,2+B)*varStrainEnergy[2+B]);
423 }
424 jacobian[B][3] = -(1./yield.grad(varYield,4) - (varIsoHardening[0] - alpha_TR)*pow(yield.grad(varYield,4),-2.)*
425 yield.hess(varYield,4,4)*harden.grad(varIsoHardening,0))*yield.grad(varYield,B);
426 //This equation is half of the log of equation(26)
427 residual(B) = 0.5*log(eigTR(B)) - gamma_Dt*yield.grad(varYield,B) - log(varStrainEnergy[2+B]);
428 }
429 jacobian[3][3] = yield.grad(varYield,4)*harden.grad(varIsoHardening,0);
430 jacobian[4][3] = -yield.hess(varYield,4,4)*harden.grad(varIsoHardening,0);
431 jacobian[4][4] = 1;
432
433 //Equation (30)
434 residual(3) = yield(varYield);
435 residual(4) = zeta - yield.grad(varYield,4);
436
437 //Invert jacobian matrix
438 jacobian.gauss_jordan();
439 //Multiply the jacobian inverse and the function, the result is "product"
440 jacobian.vmult(product,residual);
441
442 //Update Newton-Raphson variables
443 for(unsigned int i = 0; i < dim; i++){
444 //Update lambda_e(i)
445 varStrainEnergy[2+i] -= product(i);
446 }
447 //Update alpha
448 varIsoHardening[0] -= product(3);
449 //Update zeta
450 zeta -= product(4);
451
452 //Update other variables
453 for(unsigned int i = 0; i < dim; i++){
454 //Update beta, equation (9)
455 varYield[i] = strain_energy.grad(varStrainEnergy,2+i)*varStrainEnergy[2+i]; //beta(i)
456 }
457 //Update q, directly from the hardening function
458 varYield[4] = harden(varIsoHardening);
459 //Update gamma_Dt, based on equation (25). Pull in a little from the yield surface for stability.
460 gamma_Dt = (varIsoHardening[0] - alpha_TR)/zeta;
461 //Update xi
462 for(unsigned int i=0; i<dim; i++){
463 nuBar(i) = (log(varStrainEnergy[2+i]) - 0.5*log(eigTR(i)))/(-gamma_Dt);
464 varYield[dim+2+i] = xi_TR(i) + 2./3.*gamma_Dt*properties.H*nuBar(i);
465 }
466
467 //Check for convergence
468 if(product.l2_norm() < Tolerance || residual.l2_norm() < Tolerance){
469 break;
470 }
471 if(iter>30){
472 this->pcout << " During update of plastic variables: Maximum number of iterations reached without convergence. \n";
473 this->pcout << " Consider using a smaller load or a higher number of increments. \n";
474 if (stopOnConvergenceFailure) {exit (1);}
475 else {this->pcout << " stopOnConvergenceFailure==false, so marching ahead\n";}
476 break;
477 }
478 iter++;
479 }
480 //if(cellID == 0)
481 // std::cout << "Before: " << yield(varYield) << " " << gamma_Dt << std::endl;
482//*
483 //For stability, pull in a little from the yield surface. Recalculate the updated
484 //variables with this adjusted value for gamma_Dt.
485 gamma_Dt *= 1 - 1.e-8;
486
487 //Update alpha, equation (25)
488 alpha = alpha_TR + gamma_Dt*yield.grad(varYield,4);
489 varIsoHardening[0] = alpha;
490 //Update the isotropic hardening, q
491 varYield[4] = harden(varIsoHardening);
492 for(unsigned int i = 0; i < dim; i++){
493 //Update the principal stresses
494 beta(i) = varYield[i];
495 //Update the derivative of the yield function, f w.r.t. beta
496 nuBar(i) = yield.grad(varYield,i);
497 //Update the principal elastic stretches, equation (26)
498 lambda_e(i) = exp(-gamma_Dt*nuBar(i))*sqrt(eigTR(i));
499 varStrainEnergy[2+i] = lambda_e(i);
500 //Update the back stress, xi
501 xi(i) = xi_TR(i) + 2./3.*gamma_Dt*properties.H*nuBar(i);
502 varYield[dim+2+i] = xi(i);
503 }//*/
504
505 //if(cellID == 0)
506 // std::cout << "After: " << yield(varYield) << " " << gamma_Dt << std::endl;
507
508 }
509 //Find tau, equation (29)
510 tau.equ(beta(0),eigDyad[0],beta(1),eigDyad[1],beta(2),eigDyad[2]);
511 //Find b_e, equation (28)
512 b_e.equ(std::pow(lambda_e(0),2),eigDyad[0],std::pow(lambda_e(1),2),eigDyad[1],std::pow(lambda_e(2),2),eigDyad[2]);
513 //Find invCP, based on equation (4)
514 b_e.mTmult(be_FinvT,F_inv); //temporary variable
515 F_inv.mmult(invCp,be_FinvT); //invCp = F_inv*b_e*F_inv^T
516
517 //Store history variables for this iteration
518 histInvCP_iter[cellID][quadPtID] = invCp;
519 histAlpha_iter[cellID][quadPtID] = alpha;
520 histXi_iter[cellID][quadPtID] = xi;
521
522 //Store the von Mises stress to project to the nodes and include in output file
523 projectVonMisesStress[cellID][quadPtID] = std::sqrt(0.5*(std::pow(beta(0) - beta(1),2.) +
524 std::pow(beta(1) - beta(2),2.) +
525 std::pow(beta(2) - beta(0),2.)));
526
527 //Determine algorithmic elastoplastic tangent (section 4 of the formulation)
528 FullMatrix<double> a_ep(dim,dim), a_e(dim,dim);
529 for (unsigned int A=0; A<dim; ++A){
530 for (unsigned int B=0; B<dim; ++B){
531 //Equation (36)
532 a_e[A][B] = strain_energy.hess(varStrainEnergy,2+A,2+B)*varStrainEnergy[2+A]*varStrainEnergy[2+B] +
533 strain_energy.grad(varStrainEnergy,2+A)*varStrainEnergy[2+A]*(A==B);
534 }
535 }
536 if(yield_TR <= 0){
537 //If elastic...
538 a_ep = a_e;
539 }
540 else{
541 //If plastic...
542
543 //The next several lines go into equation (35)
544 FullMatrix<double> a_eInv(dim,dim), f2_B(dim,dim);
545 a_eInv.invert(a_e);
546 //Find the 2nd derivatives of f w.r.t. beta
547 for (unsigned int A=0; A<dim; ++A){
548 for (unsigned int B=0; B<dim; ++B){
549 f2_B[A][B] = yield.hess(varYield,A,B);
550 }
551 }
552
553 FullMatrix<double> h1(dim,dim), h2(dim,dim), h3(dim,dim), h3h2(dim,dim), h2h3h2(dim,dim), h2_etc(dim,dim);
554 double h4;
555 //Find h1
556 h1.equ(1,Identity,2./3.*properties.H*gamma_Dt,f2_B);
557 h1.gauss_jordan();
558 //Find h2
559 h1.mmult(h2,f2_B); //h1*f2_B
560 h2.equ(1,Identity,-2./3.*properties.H*gamma_Dt,h2);
561 //Find h3
562 f2_B.mmult(h3,h2); //f2_B*h2
563 h3.equ(1.,a_eInv,gamma_Dt,h3);
564 h3.gauss_jordan();
565 //Find h4
566 h4 = 1. - gamma_Dt*harden.grad(varIsoHardening,0)*yield.hess(varYield,4,4);
567 //Find h3*h2
568 h3.mmult(h3h2,h2);
569 //Find h2*h3*h2
570 h2.mmult(h2h3h2,h3h2);
571 //Find h2*h3*h2 + 2/3*H*h1
572 h2_etc.equ(1,h2h3h2,2./3.*properties.H,h1);
573
574 Vector<double> h3h2_nuBar(dim), h2etc_nuBar(dim);
575 //Temporary variables, recall that nuBar is the partial of f w.r.t. beta
576 h3h2.vmult(h3h2_nuBar,nuBar);
577 h2_etc.vmult(h2etc_nuBar,nuBar);
578
579 //The next two lines are equation (34)
580 a_ep.outer_product(h3h2_nuBar,h3h2_nuBar);
581 a_ep.equ(1.,h3,-h4/(h4*(nuBar*h2etc_nuBar) - pow(yield.grad(varYield,4),2.)*harden.grad(varIsoHardening,0)),a_ep);
582
583 }
584 for (unsigned int i=0; i<dim; ++i){
585 for (unsigned int j=0; j<dim; ++j){
586 for (unsigned int k=0; k<dim; ++k){
587 for (unsigned int l=0; l<dim; ++l){
588 if(this->currentIncrement_cp == 0 && this->currentIteration==0){
589 //For readability, extract the Lame parameters
590 double lambda = varStrainEnergy[0], mu = varStrainEnergy[1];
591 //c_{ijkl}=C_{IJKL}, the standard modulus, in the limit as stretches go to one
592 //This is done to avoid dividing by zero.
593 c[i][j][k][l] = lambda*(i==j)*(k==l) + mu*((i==k)*(j==l) + (i==l)*(j==k));
594 }
595 else{
596 //These two for loops are computing equations (33) and (38) of the formulation
597 for (unsigned int A=0; A<dim; ++A){
598 for (unsigned int B=0; B<dim; ++B){
599 c[i][j][k][l] += a_ep[A][B]*eigDyad[A][i][j]*eigDyad[B][k][l];
600 }
601 c[i][j][k][l] += 2.*beta(A)/d_A(A)*(0.5*(b_eTR[i][k]*b_eTR[j][l] + b_eTR[i][l]*b_eTR[j][k])
602 - b_eTR[i][j]*b_eTR[k][l]
603 - b_eTR.determinant()/eigTR(A)*(0.5*((i==k)*(j==l) + (i==l)*(j==k))
604 - ((i==j) - eigDyad[A][i][j])*((k==l) - eigDyad[A][k][l]))
605 + eigTR(A)*(b_eTR[i][j]*eigDyad[A][k][l]

617template <int dim>
618void continuumPlasticity<dim>::getElementalValues(FEValues<dim>& fe_values,
619 unsigned int dofs_per_cell,
620 unsigned int num_quad_points,
621 FullMatrix<double>& elementalJacobian,
622 Vector<double>& elementalResidual)
623{
624
625 //Initialized history variables and pfunction variables if unititialized
626 if(initCalled == false){
627 init(num_quad_points);
628 }
629
630 //Get the element number.
631 unsigned int cellID = fe_values.get_cell()->user_index();
632 //Reset the vectors and matrices in static condensation to zero.
633 //This is necessary because we are using an enhanced strain (see the paper
634 //cited in the formulation).
635 enhStrain.staticCondensationData[cellID].reset();
636
637 //Vector relating local to global degree of freedom numbers
638 std::vector<types::global_dof_index> local_dof_indices(dofs_per_cell);
639 //Local displacement vector
640 Vector<double> Ulocal(dofs_per_cell);
641
642 //Initiate element iterator
643 typename DoFHandler<dim>::active_cell_iterator cell(& this->triangulation,
644 fe_values.get_cell()->level(),
645 fe_values.get_cell()->index(),
646 & this->dofHandler);
647 //Pass in the element number
648 cell->set_user_index(fe_values.get_cell()->user_index());
649 //Populate the local to global dof vector
650 cell->get_dof_indices (local_dof_indices);
651
652 //Populate the locacal displacement vector
653 for(unsigned int i=0; i<dofs_per_cell; i++){
654 Ulocal[i] = this->solutionWithGhosts[local_dof_indices[i]];
655 }
656 //Initialize the enhanced strain object with this information.
657 enhStrain.reinit(Ulocal, cell);
658
659 //loop over quadrature points
660 for (unsigned int q=0; q<num_quad_points; ++q){
661 //Get enhanced deformation gradient
662 enhStrain.get_F_enh(q, F);
663
664 //Update strain, stress, and tangent for current time step/quadrature point
665 calculatePlasticity(cellID, q);
666
667 //Update block matrices and vectors in enhanced strain
668 enhStrain.create_block_mat_vec(F, tau, c, q);
669
670 //Pass local matrices and vectors to static condensation
671 //Original dofs matrix
672 enhStrain.staticCondensationData[cellID].K.add(enhStrain.fe_values.JxW(q),enhStrain.Klocal);
673 //Cross terms matrix
674 enhStrain.staticCondensationData[cellID].G.add(enhStrain.fe_values.JxW(q),enhStrain.Glocal);
675 //Enhanced dofs matrix
676 enhStrain.staticCondensationData[cellID].M.add(enhStrain.fe_values.JxW(q),enhStrain.Mlocal);
677 //Original dofs vector
678 enhStrain.staticCondensationData[cellID].F.add(enhStrain.fe_values.JxW(q),enhStrain.Flocal);
679 //Enhanced dofs vector
680 enhStrain.staticCondensationData[cellID].H.add(enhStrain.fe_values.JxW(q),enhStrain.Hlocal);
681 }
682 //Perform static condensation
683 enhStrain.staticCondensationData[cellID].staticCondense();
684 //Retrieve the element level jacobian
685 elementalJacobian = enhStrain.staticCondensationData[cellID].K2;
686 //Retrieve the element level residual
687 elementalResidual.equ(-1.,enhStrain.staticCondensationData[cellID].F2);
688}

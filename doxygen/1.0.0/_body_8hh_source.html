41 void read_vtk( const std::string &vtkfile)
42 {
43 std::cout << "Begin reading vtk file" << std::endl;
44
45
46 // read in vtk file here
47 std::ifstream infile_mesh(vtkfile.c_str());
48
49 // read mesh info
50 mesh.read_vtk(infile_mesh);
51
52 std::ifstream infile(vtkfile.c_str());
53
54 // read point data
55 std::istringstream ss;
56 std::string str, name, type, line;
57 int numcomp;
58 unsigned long int Npoints;
59
60 while(!infile.eof())
61 {
62
63 std::getline( infile, line);
64
65 if( line[0] == 'P')
66 {
67 if( line.size() > 9 && line.substr(0,10) == "POINT_DATA")
68 {
69 //std::cout << line << "\n";
70 ss.clear();
71 ss.str(line);
72 ss >> str >> Npoints;
73
74 }
75 }
76 else if( line[0] == 'S')
77 {
78 if( line.size() > 6 && line.substr(0,7) == "SCALARS")
79 {
80 ss.clear();
81 ss.str(line);
82 ss >> str >> name >> type >> numcomp;
83
84 // read LOOKUP_TABLE line
85 std::getline( infile, line);
86
87 // read data
88 std::cout << "begin reading data" << std::endl;
89 std::vector<double> data(Npoints);
90 for( unsigned int i=0; i<Npoints; i++)
91 {
92 infile >> data[i];
93 //std::cout << data[i] << std::endl;
94 }
95 std::cout << " done" << std::endl;
96
97
98 // construct field
99 std::vector<std::string> var_name(DIM);
100 std::vector<std::string> var_description(DIM);
101
102 if( DIM >= 2)
103 {
104 var_name[0] = "x";
105 var_description[0] = "x coordinate";
106 var_name[1] = "y";
107 var_description[1] = "y coordinate";
108
109 }
110 if( DIM >= 3)
111 {
112 var_name[2] = "z";
113 var_description[2] = "z coordinate";
114
115 }
116
117
118 std::cout << "Construct PField '" << name << "'" << std::endl;
119 scalar_field.push_back( PField<Coordinate, double, DIM>( name, var_name, var_description, mesh, data, 0.0) );
120 std::cout << " done" << std::endl;
121
122 }
123 }
124 // Alternative field descriptor used by ParaView (holds the same information as the "SCALAR" line above)
125 else if( line[0] == 'F')
126 {
127 if( line.size() > 14 && line.substr(0,15) == "FIELD FieldData")
128 {
129 ss.clear();
130 ss.str(line);
131 ss >> str >> numcomp;
132
133 // read LOOKUP_TABLE line
134 std::getline( infile, line);
135
136 ss.clear();
137 ss.str(line);
138 ss >> name >> numcomp >> Npoints >> type;
139
140
141 // read data
142 std::cout << "begin reading data" << std::endl;
143 std::vector<double> data(Npoints);
144 for( unsigned int i=0; i<Npoints; i++)
145 {
146 infile >> data[i];
147 }
148 std::cout << " done" << std::endl;
149
150
151 // construct field
152 std::vector<std::string> var_name(DIM);
153 std::vector<std::string> var_description(DIM);
154
155 if( DIM >= 2)
156 {
157 var_name[0] = "x";
158 var_description[0] = "x coordinate";
159 var_name[1] = "y";
160 var_description[1] = "y coordinate";
161
162 }
163 if( DIM >= 3)
164 {
165 var_name[2] = "z";
166 var_description[2] = "z coordinate";
167
168 }
169
170
171 std::cout << "Construct PField '" << name << "'" << std::endl;
172 scalar_field.push_back( PField<Coordinate, double, DIM>( name, var_name, var_description, mesh, data, 0.0) );
173 std::cout << " done" << std::endl;
174
175 }
176 }
177 }
178
179 infile.close();
